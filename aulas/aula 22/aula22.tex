\begin{frame}[fragile]
  \frametitle{Triângulos - programando}

  Escreva uma função que calcule o número de triângulos
  em pé de uma grade de tamanho $n$.

  \pause

  \begin{exampleblock}{}
\begin{verbatim}
int triangulos(int n) {
  if (n == 0)
      return 0;
  else if (n == 1)
      return 1;
  else
      return n + 2*triangulos(n-1) - triangulos(n-2);
}\end{verbatim}
  \end{exampleblock}

  \pause

  Observe que a função \cod{triangulos} chama a própria
  função \cod{triangulos}. {\bf Isso é chamado de \alert{recursão}}.

\end{frame}

\section{Recursão}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}[fragile]
  \frametitle{Recursão}

  \begin{center}
    \includegraphics{matrioshka.jpg}
  \end{center}

  \begin{block}{Recursão}
    A ideia é que um problema pode ser resolvido da
    seguinte maneira:

    \begin{itemize}
      \pause
      \item \alert{Primeiramente}, definimos as soluções para casos básicos.
      \pause
      \item \alert{Em seguida}, tentamos reduzir o problema
        para instâncias menores.
      \pause
      \item \alert{Finalmente}, combinamos o resultado das instâncias
        menores para obter um resultado do problema original.
    \end{itemize}
  \end{block}
\end{frame}


%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}[fragile]
  \frametitle{Exemplo - Fatorial}

  Certos problemas são naturalmente recursivos.

  \begin{block}{Problema}
    O fatorial de $n$ é definido como:
    $$
      n! =
      \begin{cases}
        1               &      \mbox{se } n = 0 \\
        n \cdot (n-1)!  &      \mbox{se } n > 0.
      \end{cases}
    $$
    Escreva um programa para calcular o valor do fatorial de um número.
  \end{block}

  \pause Qual é a \alert{base} da recursão? \\
  \pause {\bf Resposta:} o caso $n = 0$.

  \vspace*{0.2cm}

  \pause A que \alert{instância menor} nós reduzimos o problema? \\
  \pause {\bf Resposta:} para a instância $(n - 1)!$.

  \vspace*{0.2cm}

  \pause Como nós \alert{combinamos} o resultado para resolver o problema original? \\
  \pause {\bf Resposta:} multiplicando por $n$ o resultado do problema menor.
\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\begin{frame}[fragile]
  \frametitle{Exemplo - Fatorial}


  \begin{exampleblock}{}
    \begin{semiverbatim}
  int fatorial(int n) \{
      int x, y;

      \alert{// Caso base}
      if (n == 0) \{
          return 1;

      \alert{// Caso geral}
      \} else \{
          \alert{// reduzimos o problema para instância menor x}
          x = n - 1;
          y = fatorial(x);

          \alert{// combinamos o resultado y da instância menor}
          return n * y;
      \}
  \}\end{semiverbatim}
  \end{exampleblock}

  \begin{verbatim}
  \end{verbatim}
\end{frame}


\section{Pilha de chamadas}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}
  \frametitle{Pilha de chamadas}

  \begin{block}{Lidando com variáveis da função}
    \begin{itemize}
      \item Toda função tem suas próprias variáveis locais

      \pause

      \item {\bf Então}, cada chamada da função \cod{fatorial}
          cria as variáveis $n$, $x$ e $y$.

      \vspace*{0.4cm}
      \pause

      \item Mas chamamos a função fatorial várias vezes:
          \alert{\cod{fatorial(n)}}, \alert{\cod{fatorial(n-1)}}, ...,
          \alert{\cod{fatorial(1)}} e \alert{\cod{fatorial(0)}}!

      \pause

      \item {\bf Então}, em um dado instante, podem existir várias
          variáveis $n$, $x$ e $y$, um trio para cada chamada.
    \end{itemize}
  \end{block}

  \pause

  \begin{alertblock}{Pilha de chamadas}
    Para manter várias cópias de variáveis locais de uma função,
    elas são colocadas em uma {\bf pilha de chamada} (ou {\bf pilha de execução})
    toda vez que uma função é chamada e são retiradas da pilha quando a função termina.
  \end{alertblock}

\end{frame}

  \begin{block}{Problema}
    A torre de Hanói é um brinquedo com três estacas A, B e C e
