\begin{frame}[fragile]
  \frametitle{Ordenação}

  \begin{block}{Ordenação}
    \begin{itemize}
      \item Vamos estudar algoritmos para ordenar conjuntos de elementos
      \pause
      \item Os elementos podem ser de qualquer tipo que possamos comparar:
      \pause
      \begin{itemize}
        \item números inteiros,
        \pause
        \item nomes de pessoas,
        \pause
        \item times de futebol... :)
      \end{itemize}
      \pause
      \item Os algoritmos podem ordenar \alert{crescente}  ou
        \alert{decrescentemente}, dependendo da direção da comparação.
    \end{itemize}
  \end{block}

  \pause

  \begin{block}{Estratégias}
    Existem várias estratégias para ordenar:
    \pause
    \begin{itemize}
      \item Selecionar o menor a cada vez e colocar na ponta
      \pause
      \item Trocar itens fora de ordem
      \pause
      \item Outras?
    \end{itemize}
    \pause
    Estratégias diferentes levam a algoritmos diferentes.
  \end{block}
\end{frame}

\begin{frame}[fragile]
  \frametitle{Convenções}

  \begin{block}{Algoritmo de ordenação}
    \begin{itemize}
      \item Vamos implementar uma função para ordenar inteiros
      \pause
      \item A função terá os seguintes parâmetros:
      \pause
      \begin{itemize}
        \item \alert{\cod{int vetor[]}}: vetor de inteiros onde os elementos estão
        \pause
        \item \alert{\cod{int n}}: o número de elemento do vetor
      \end{itemize}
      \pause
      \item A função deverá ordenar o vetor passado \alert{crescentemente}
    \end{itemize}
  \end{block}

  \pause

  Para trocar dois valores, vamos sempre usar a seguinte função:
\begin{exampleblock}{Trocar dois valores inteiros}
\begin{verbatim}
void trocar(int *a, int *b) {
    int aux = *a;
    *a = *b;
    *b = aux;
}\end{verbatim}
  \end{exampleblock}
\end{frame}

\begin{frame}[fragile]
  \frametitle{Algoritmo de ordenação por seleção ({\it Selection-Sort})}

  \begin{exampleblock}{Menor elemento não ordenado (na lista preta)}
    \small
\begin{verbatim}
int menor_elemento(int vetor[], int n, int primeiro) {
    int i, menor = primeiro;
    for (i = primeiro + 1; i < n; i++) {
        if (vetor[i] < vetor[menor])
            menor = i;
    }
    return menor;
}\end{verbatim}
  \end{exampleblock}

  \pause

  \begin{exampleblock}{Ordenação por seleção}
    \small
\begin{verbatim}
int ordenar_selecao(int vetor[], int n) {
    int i, menor;
    for (i = 0; i < n; i++) {
        menor = menor_elemento(vetor, n, i);
        trocar(&vetor[i], &vetor[menor]);
    }
}\end{verbatim}
  \end{exampleblock}
\end{frame}

\begin{frame}[fragile]
  \frametitle{Algoritmo de ordenação por inserção ({\it Insertion-Sort})}

  \begin{exampleblock}{Posição de inserção (na lista verde)}
    \small
\begin{verbatim}
int posicao_elemento(int vetor[], int ultimo, int elemento) {
    int i;
    for (i = 0; i <= ultimo && vetor[i] <= elemento; i++);
    return i;
}\end{verbatim}
  \end{exampleblock}

  \pause

  \begin{exampleblock}{Deslocar parte do vetor}
    \small
\begin{verbatim}
void deslocar_subvetor(int vetor[], int primeiro, int ultimo) {
    int i;
    for (i = ultimo; i >= primeiro; i--) {
        vetor[i+1] = vetor[i];
    }
}\end{verbatim}
  \end{exampleblock}

\end{frame}

\begin{frame}[fragile]
  \frametitle{Algoritmo de ordenação por inserção ({\it Insertion-Sort})}

  \begin{exampleblock}{Ordenação por inserção}
\begin{verbatim}
int ordenar_insercao(int vetor[], int n) {
    int i, posicao;
    int elemento;

    for (i = 1; i < n; i++) {
        elemento = vetor[i];
        posicao = posicao_elemento(vetor, i-1, elemento);
        deslocar_subvetor(vetor, posicao, i-1);
        vetor[posicao] = elemento;
    }
}\end{verbatim}
  \end{exampleblock}
\end{frame}

void ordenar_bolha(int vetor[], int n) {
    int i, j;
    for (i = n - 1; i > 0; i--) {
        for (j = 0; j < i; j++) {
            if (vet[j] < vet[j+1])
                trocar(&vetor[j], &vetor[j+1]);

    }
}
\end{verbatim}
