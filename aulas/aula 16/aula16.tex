\section {Vetores}

\begin{frame}
  \frametitle{Vetores --- Definição}

  \begin{alertblock}{Vetor}
    Vetor é uma coleção de variáveis de um mesmo tipo
    referenciada por um nome comum.
  \end{alertblock}

  \pause

  \begin{block}{Características}
  \begin{itemize}
    \item Cada variável é acessada por por meio de índice
    \pause
    \item O tamanho máximo $n$ da coleção é pré-definido
        (deve ser constante)
    \pause
    \item Os índices \alert{devem} estar em um intervalo definido,
        normalmente de $0$ a $n-1$ ou de $1$ a $n$.
  \end{itemize}
 \end{block}
\end{frame}

\begin{frame}[fragile]

  \frametitle{Declaração de um vetor}

  \begin{block}{Vetor de tamanho $N$}
    \alert{$<$tipo$>$} identificador[\alert{$N$}];
  \end{block}

  \pause

  {\bf Em C:}
  \begin{itemize}
    \item A primeira variável do vetor tem índice \alert{$0$}
    \item e a última variável do vetor tem índice \alert{$N - 1$}.
  \end{itemize}

  \pause

  \begin{exampleblock}{Exemplo}
\begin{verbatim}
float notas[100];
int medias[100];
char nome[200];\end{verbatim}
  \end{exampleblock}
\end{frame}


\begin{frame}[fragile]

  \frametitle{Usando um vetor}

  \begin{block}{Acessando a variável na posição $i$}
    a = vetor[\alert{$i$}];
  \end{block}

  \pause

  \begin{itemize}
    \item A posição $i$ pode ser qualquer valor inteiro
    \item Podemos usar constantes ou variáveis para a posição
  \end{itemize}

  \pause

  \begin{exampleblock}{Exemplo}
  \small
\begin{verbatim}
float notas[10], ultima_nota;
int i;

// todos tiraram nota 8.5
for(i = 0; i < 10; i++) {
  notas[i] = 8.5;
}

// o último aluno ganhou um ponto de bônus
notas[9] = notas[9] + 1.0;
ultima_nota = notas[9];\end{verbatim}
  \end{exampleblock}
\end{frame}

\begin{frame}
  \frametitle{Vetores - Melhores práticas}

  \begin{block}{Cuidados}
    \begin{itemize}
      \item O tamanho do vetor é constante (não pode ser mudado durante a execução)
      \pause
      \item {\bf Então}, escolher sempre um bom tamanho para o vetor
          \pause
          \begin{itemize}
            \item Se for muito \alert{grande}, haverá desperdício de memória
            \item Se for muito \alert{pequeno}, faltará espaço para os dados
          \end{itemize}
      \bigskip
      \pause
      \item Sempre verificar duas vezes os índices dos vetores
      \pause
      \item {\bf Porque} índices fora dos limites causam efeitos indesejados
        \pause
        \begin{itemize}
          \item Alteram valores de outras variáveis
          \item Causam \alert{segmentation fault} (tentativa de acesso a
            um endereço de memória que não existe)
        \end{itemize}
    \end{itemize}
  \end{block}
\end{frame}

\begin{frame}[t,fragile]
  \frametitle{Vetores - Representação na memória}

  \begin{center}
    \onslide<2-> {
      \begin{tabular}{|c|c|c|c|c|c|c|c|}
        \hline
        Nome & d & \multicolumn{5}{|c|}{vetor} & f \\
        \hline
        Índice & - & 0 & 1 & 2 & 3 & 4 & - \\
        \hline
        &\only<1-5>{\onslide<3->{0}}\only<6->{1}& & & & \onslide<4->{9} & & \onslide<5->{5} \\
        \hline
      \end{tabular}
    }
  \end{center}

  Vamos executar o seguinte código:

  \begin{exampleblock}{}
\begin{semiverbatim}
\alert<2>{int d, vetor[5], f;}
\alert<3>{d = 0;             }
\alert<4>{vetor[3]  = 9;     }
\alert<5>{vetor[5]  = 5;     }
\alert<6>{vetor[-1] = 1;     }
\alert<7>{printf("\%d", d);  } \end{semiverbatim}
  \end{exampleblock}

  \begin{itemize}
    \item<8->O programa irá imprimir 1!
    \item<9->Dependendo de como o compilador organiza a memória,
        o programa pode sair com {\it segmentation fault}
  \end{itemize}

\end{frame}

\begin{frame}[fragile]
  \frametitle{Parênteses: definindo constantes}

  Quantas vezes aparece o 60?
  E se o número de alunos mudar?

  \begin{exampleblock}{}
  \small
\begin{verbatim}
#include <stdio.h>

int main() {
    float provas[60], exercicios[60], media_turma;
    int i;

    printf("Notas das provas e exercícios dos alunos: ");
    for (i = 0; i < 60; i++)
        scanf("%f %f", &provas[i], &exercicios[i]);

    for (media_turma = 0, i = 0; i < 60; i++)
        media_turma += (provas[i] + exercicios[i])/2;
    media_turma = media_turma / 60;

    printf("Media da turma: %f\n", media_turma);
}\end{verbatim}
  \end{exampleblock}


\end{frame}

\begin{frame}[fragile]
  \frametitle{Parênteses: definindo constantes}

  Podemos dar um nome uma constante! Basta usar a diretiva
  \alert{\cod{\#define}}.

  \begin{exampleblock}{}
  \small
\begin{verbatim}
#include <stdio.h>

#define NUM_ALU 60

int main() {
    float provas[NUM_ALU], exercicios[NUM_ALU], media_turma;
    int i;

    printf("Notas das provas e exercícios dos alunos: ");
    for (i = 0; i < NUM_ALU; i++)
        scanf("%f %f", &provas[i], &exercicios[i]);

    for (media_turma = 0, i = 0; i < NUM_ALU; i++)
        media_turma += (provas[i] + exercicios[i])/2;
    media_turma = media_turma / NUM_ALU;

    printf("Media da turma: %f\n", media_turma);
}\end{verbatim}
  \end{exampleblock}

\end{frame}
