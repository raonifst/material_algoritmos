\section{Expressões lógicas}

\begin{frame}

  \frametitle{Expressões lógicas} 
  
  Expressões lógicas são aquelas que realizam uma operação lógica
  ({\tt ou}, {\tt e}, {\tt não}, etc...) e retornam verdadeiro ou falso
  (como as expressões relacionais).

\end{frame}

\begin{frame}

\frametitle{Operadores Lógicos}

  \begin{block}{Os operadores lógicos são}
    \begin{itemize}
    \item \alert{\cod{\&\&}} : operador E
    \item \alert{\cod{||}} : operador OU
    \item \alert{\cod{!}} : operador NÃO 
    \end{itemize}   
  \end{block}



\end{frame}


\begin{frame}
  \frametitle{Expressões lógicas}

  \begin{itemize}

  \item $<$expressao$>$ \alert{\tt \&\&} $<$expressao$>$: Retorna verdadeiro quando ambas
    as expressões são verdadeiras. Sua tabela verdade é:

    \begin{block}{}
      \begin{center}
        \begin{tabular}{|c|c|c|}
          \hline
          $Op_1$&$Op_2$&$Ret$\\
          \hline
          V&V&V\\
          V&F&F\\
          F&V&F\\
          F&F&F\\
          \hline
        \end{tabular}
      \end{center}
    \end{block}
    
    \begin{block}{Exemplo}
      \cod{a == 0 \&\& b == 0}
    \end{block}
  \end{itemize}
\end{frame}


\begin{frame}
  \frametitle{Expressões lógicas}

  \begin{itemize}

  \item $<$expressao$>$ \alert{\tt ||} $<$expressao$>$: Retorna verdadeiro
    quando pelo menos uma das expressões é verdadeiras. Sua tabela
    verdade é:

    \begin{block}{}
      \begin{center}
        \begin{tabular}{|c|c|c|}
          \hline
          $Op_1$&$Op_2$&$Ret$\\
          \hline
          V&V&V\\
          V&F&V\\
          F&V&V\\
          F&F&F\\
          \hline
        \end{tabular}
      \end{center}
    \end{block}
    
    \begin{block}{Exemplo}
      \cod{a == 0 || b == 0}
    \end{block}
  \end{itemize}
\end{frame}


\begin{frame}
  \frametitle{Expressões lógicas}

  \begin{itemize}

  \item \alert{\tt !} $<$expressao$>$: Retorna verdadeiro quando a expressão
    é falsa e vice-versa. Sua tabela verdade é:

    \begin{block}{}
      \begin{center}
        \begin{tabular}{|c|c|}
          \hline
          $Op_1$&$Ret$\\
          \hline
          V&F\\
          F&V\\
          \hline
        \end{tabular}
      \end{center}
    \end{block}
    
    \begin{block}{Exemplo}
      \cod{!(a == 0)}
    \end{block}
  \end{itemize}
\end{frame}


\begin{frame}
  \frametitle{Simplificações úteis}

  \begin{block}{Equivalências}
  \begin{itemize}
    \item {\tt !(a == b)} é equivalente a {\tt a != b}
    \item {\tt !(a != b)} é equivalente a {\tt a == b}
    \item {\tt !(a > b)} é equivalente a {\tt a <= b}
    \item {\tt !(a < b)} é equivalente a {\tt a >= b}
    \item {\tt !(a >= b)} é equivalente a {\tt a < b}
    \item {\tt !(a <= b)} é equivalente a {\tt a > b}
  \end{itemize}
  \end{block}
\end{frame}

\section{Mais Comandos condicionais}

\begin{frame}[fragile]

  \frametitle{Combinando comandos condicionais}

Às vezes podemos simplificar
\begin{exampleblock}{}
\begin{verbatim}
if (cond1) {
  if (cond2) {
    if (!cond3) {
      comando;
    }
  }
}
\end{verbatim}
\end{exampleblock}

por

\begin{exampleblock}{}
\begin{verbatim}
if (cond1 && cond2 && !cond3) {
  comando;
}
\end{verbatim}
\end{exampleblock}

\end{frame}

\begin{frame}[fragile]
  \frametitle{if-else-if Encaixados}

\begin{itemize}
\item Várias vezes a construção if-else-if alterna entre valores \alert{inteiros}:
\end{itemize}
\begin{exampleblock}{}
\begin{verbatim}
#include <stdio.h>
int main () {
    long rga;
    scanf("%ld", &rga);
    if (rga == 201421901004) 
        printf("Abel Breno Ventura de Abreu\n");
    else if (rga == 201421904008)
        printf("Ana Flavia Rosa Manzano\n"); 
    else printf("Nenhum aluno com RGA informado!");             
    
    return 0;
}
\end{verbatim}
\end{exampleblock}
\end{frame}


\begin{frame}
  
  \frametitle{O comando \texttt{switch} } 
  
  \begin{itemize}
    
  \item O objetivo do comando {\tt switch} é simplificar uma expressão
    onde as condições ocorrem sobre uma variável \alert{inteira} (pode ser \alert{caractere}): 

    \vspace{20pt}

    \begin{block}{Sintaxe}
    
    {\tt switch (variável inteira) \{
      
      \hspace{16pt} case valor1:

      \hspace{32pt} {\sl comandos} 
      
      \hspace{32pt} break;
      
      \hspace{16pt} case valor2: 

      \hspace{32pt} {\sl comandos} 
      
      \hspace{32pt} break;

      \}
    }

    \end{block}

  \end{itemize}
  
\end{frame}

\begin{frame}[fragile]

  \frametitle{O comando \texttt{switch} } 


\begin{block}{}
\begin{verbatim}
 switch(rga) {
    case 201421901004: 
       printf("Abel Breno Ventura de Abreu\n");
       break;
    case 201421904008:
        printf("Ana Flavia Rosa Manzano\n");
        break;
    
  }
\end{verbatim}
\end{block}
\end{frame}

\begin{frame}

  \frametitle{O comando {\tt switch} } 
  
  \begin{itemize}
    
  \item Os comandos começam a ser executados a partir do 
    ponto onde o valor da variável corresponde ao valor antes dos dois
    pontos (:).

  \item Executa todos os comandos até que encontre um comando {\tt break}
    ou que chegue ao final do bloco de comandos do {\tt switch}

  \end{itemize}
  
\end{frame}

\begin{frame}

  \frametitle{Valor padrão} 
  
  \begin{itemize}
    
  \item Você pode utilizar uma condição {\tt default}.
    A execução dentro da alternativa {\tt default} ocorre se nenhuma outra
    condição foi verdadeira (assim como o último else do if-else-if encaixados). 

    \vspace{20pt}

    \begin{block}{Sintaxe do {\tt switch}}
    
    {\tt \alert{switch} (variável inteira) \{
      
      \hspace{16pt} case valor: {\sl comandos} break;
    
      \hspace{16pt} \alert{default:} {\sl comandos} 
     
      \}
    }

    \end{block}

  \end{itemize}
  
\end{frame}

\begin{frame}[fragile]

  \frametitle{Valor padrão} 

\begin{exampleblock}{}
\begin{verbatim}
   switch(rga) {
    case 201421901004: 
       printf("Abel Breno Ventura de Abreu\n");
       break;
    case 201421904008:
        printf("Ana Flavia Rosa Manzano\n");
        break; 
      default:
          printf("O aluno não está matriculado\n");
  }
\end{verbatim}
    \end{exampleblock}
\end{frame}




\section{Legibilidade de código}

\begin{frame}
  \frametitle{Legibilidade de código}

  \begin{block}{Legibilidade}
    É a qualidade do código que permite o fácil entendimento
    por seres humanos (programadores).
  \end{block}

  \begin{block}{Importância}
    \begin{itemize}
      \item Um código é escrito uma vez e lido várias vezes.

      \item Várias pessoas podem ler e entender um mesmo programa.

      \item É mais fácil fazer alterações futuras ou aproveitar em
         outros projetos.      
    \end{itemize}
  \end{block}
 
\end{frame}

\begin{frame}
  \frametitle{Legibilidade de código}

  Para um código ser legível, ele deve ter as seguintes características.

  \begin{block}{Características}
    \begin{itemize}
      \item Comentários que resumem o que faz um \alert{conjunto} de comandos.

      \item Formatação bem definida (indentação, espaçamentos, nomes)

      \item Documentação adequada de cada programa ou algoritmo.
    \end{itemize}
  \end{block}
 
\end{frame}
