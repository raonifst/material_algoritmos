\begin{frame}
  \frametitle{Modelagem}

Modelos
\begin{itemize}
	\item Capturam a \alert{essência} de um determinado objeto.
	\item Em geral, não possuem a mesma escala do objeto.
	\item Omitem alguns \alert{detalhes} do objeto.	
	\item São limitados.
\end{itemize}  

\end{frame}	

\begin{frame}
  \frametitle{Modelo Computacional}

Objeto que:
\begin{itemize}
	\item recebe dados (entrada);
	\item armazena e recupera dados (memória);
	\item realiza opera\c{c}ões de acordo com conjuntos de instru\c{c}ões (processamento) e
	\item produz um resultado (saída).
\end{itemize}  

\end{frame}

\begin{frame}
  \frametitle{Algoritmos: defini\c{c}ão}

  	\begin{itemize}
   
  \item Uma sequência \alert{finita} de passos, \alert{precisos} e \alert{bem definidos}, para a
    realização de uma tarefa, em que detalhes e formalismos dependem do `leitor'.
  
   \item \alert{Receita}  para resolver um determinado \alert{problema}. 
  
  \end{itemize}
  \pause
  \begin{block}{Exemplo de algoritmo}
    como você descreveria os passos necessários para preparar um macarrão instantâneo?
  \end{block}

 
\end{frame}		

\begin{frame}
  \frametitle{Algoritmos: defini\c{c}ão}
	{\bf Algoritmo do macarrão instantâneo:}
	\begin{enumerate}
	\item ir até a cozinha.
	\item procurar pelas embalagens de macarrão.
	\item repetir, para cada embalagem:
		\begin{enumerate}
			\item ferver 2 copos de água em uma panela.
			\item abrir a embalagem.
			\item colocar o macarrão na panela e cozinhar por 3 minutos.
			\item mexer para soltar os fios do macarrão.
			\item retirar do fogo, deixar escorrer a água e juntar em uma travessa.
		\end{enumerate}			
	\item acrescentar azeite e sal na travessa.
	\item servir.
	\end{enumerate}
	 
\end{frame}

\subsection{Caracteriza\c{c}ão}
\begin{frame}
  \frametitle{Algoritmos: caracteriza\c{c}ão}

	\begin{block}{Recapitulando...}
	  \begin{itemize}
	  \item Sequência \alert{finita} de passos, \alert{precisos} e \alert{bem definidos}, para a resolu\c{c}ão de um 							\alert{problema}.
	   \end{itemize}
	\end{block}

	Isto implica que:
	\begin{enumerate}
	\item o número de passos é finito.
	\item o texto deve ser uma receita metódica, passo-a-passo, que inclui um passo inicial, um passo final e o passo seguinte em qualquer instante da execu\c{c}ão.
	\item cada passo contém uma instru\c{c}ão \alert{elementar} para o leitor. Por exemplo,
	\pause
	
		 \only<2>{\vspace*{0.5cm}{ ``... juntar dois ovos ...'' é elementar para um  cozinheiro  } }
		 \only<3>{\vspace*{0.5cm}{  `` ... se hoje você puder provar que a cotação do dólar vai subir 10\% no próximo mês, compre \$ 1.000,00 ...''  não é elementar para muitas pessoas :)}}
		 
	\end{enumerate}

\end{frame}

	\begin{block}{Algoritmo de Euclides}	
	\begin{enumerate}
	\item Se $M$ é igual $N$, então $MDC$, $Z$, é M (ou N); pare!

			 \item se $N$ é maior que $M$,  então substitua $N$ por ($N$-$M$) e repita a partir do passo 1
		\end{enumerate}  		
	\end{enumerate}
	\end{block}

\begin{frame}
  \frametitle{Algoritmos}
	
	\begin{block}{A parte difícil de garantir}
	\begin{itemize}
	\item Partindo de objetos  de entrada \alert{válidos}, o algoritmo \alert{termina} com a resultado \alert{correto}.
	\item Partindo de objetos  de entrada \alert{inválidos}, o algoritmo pode produzir lixo e não \alert{terminar}.
	
	\end{itemize}
	\end{block}
	
	\begin{block}{Exemplo}	
	O algoritmo para o problema do MDC sempre \alert{termina}
	\end{block}
	
	
\end{frame}

\begin{frame}
  \frametitle{Termina?}

  \begin{block}{Algoritmo de Euclides}
    \begin{enumerate}
	\item Se $M$ é igual $N$, então $MDC$, $Z$, é M (ou N); pare!
  	\item Caso contrário:
  		\begin{enumerate}
  			 \item se  $M$ é maior que  $N$,  então substitua $M$ por ($M$-$N$) e repita a partir do passo 2
			 \item se $N$ é maior que $M$,  então substitua $N$ por ($N$-$M$) e repita a partir do passo 2
		\end{enumerate}  		
	\end{enumerate}
  \end{block}

		
	\begin{block}{Exemplo:}
	Quando $M$ e $N$ são positivos, temos que:
	\begin{enumerate}
	\item a cada execu\c{c}ão $M$ ou $N$ \alert{sempre} diminuem.
	\item $M$ e $N$ sempre são positivos, pois
		\begin{itemize}
		\item $M - N$ é maior que $1$, se $M$ é maior que $N$;
		\item $N - M$ é maior que $1$, se $N$ é maior que $M$;
		\end{itemize}
	\item não podemos passar de $M$ igual à $N$ igual à $1$.
	\end{enumerate}

	Logo, o algoritmo \alert{termina}!!	
	
	\end{block}

\end{frame}

\begin{frame}
  \frametitle{Algoritmos}
	
	\begin{block}{A parte difícil de garantir - Outro exemplo}
	\begin{itemize}
	\item Imagine que você criou um algoritmo que recebe um número inteiro $N$ e procura os $N$ primeiros números perfeitos. 
	\item Atualmente, conhecemos apenas 48 números perfeitos, logo, não sabemos se o algoritmo termina quando $N=49$, por exemplo.
	
	\end{itemize}
	\end{block}
	
	\begin{block}{Defini\c{c}ão: Número perfeito}
	Um número inteiro é dito perfeito se a soma de todos os seus divisores positivos próprios (excluindo ele mesmo) é igual ao 	 próprio 	número. Por exemplo,  $6$ é um número perfeito, pois é igual à $1+2+3$.
	
	
	\end{block}
