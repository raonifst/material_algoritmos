\section{Divisão e conquista novamente}

\begin{frame}[fragile]
  \frametitle{Divisão e conquista novamente}

  \begin{block}{Quick Sort}
    \pause
    \begin{itemize}
     \item \alert{\bf Divisão:} Separamos elementos pequenos e grandes
     \pause
     \item \alert{\bf Conquista:} Ordenamos cada subvetor
    \end{itemize}
  \end{block}
  
  \pause
  
    \begin{exampleblock}{QuickSort}
    \footnotesize
\begin{verbatim}
void quick_sort(int vetor[], int ini, int fim) {
    int pos;
    
    if (ini < fim){
        pos = particionar(vetor, ini, fim);

        quick_sort(vetor, ini, pos - 1);
        quick_sort(vetor, pos, fim);
    }
}   
\end{verbatim}
  \end{exampleblock}

\end{frame}


\begin{frame}[fragile]
  \frametitle{Como particionar um vetor?}

  \begin{block}{Ideia}
    \begin{itemize}
      \pause
      \item Escolhemos um valor \alert{pivô}
      \pause
      \item Separamos o vetor em duas partes:
      \begin{enumerate}
       \item {\bf primeira}: apenas elementos menores ou iguais ao pivô
       \item {\bf segunda}: apenas elementos maiores que o pivô
      \end{enumerate}
    \end{itemize}
  \end{block}

  \pause
  
  \begin{block}{Algoritmo}
    \begin{enumerate}
     \pause 
     \item Obtemos o valor do pivô:
      \begin{itemize}
       \item escolhemos sempre o valor do último elemento
      \end{itemize}
     \pause
     \item Procuramos elementos fora de ordem:
       \begin{itemize}
        \item {\bf do início ao fim}: em busca de elementos \alert{maiores} que o pivô
        \item {\bf do fim ao início}: em busca de elementos \alert{menores ou iguais} ao pivô
       \end{itemize}
     \pause
     \item Trocamos os elementos em posições erradas
    \end{enumerate}

   
  \end{block}

\end{frame}


\begin{frame}[fragile]
  \frametitle{Algoritmo de particionamento}

  \begin{exampleblock}{Particionar vetor}
    \footnotesize
\begin{verbatim}
int particionar(int vetor[], int ini, int fim) {
    int pivo;
    
    pivo = vetor[fim];
    
    while (ini < fim) {
        while (ini < fim && vetor[ini] <= pivo)
            ini++;
        
        while (ini < fim && vetor[fim] > pivo)
            fim--;
        
        troca(&vetor[ini], &vetor[fim]);
    }
    
    return ini; // ini é a posição do primeiro elemento grande
}
\end{verbatim}
  \end{exampleblock}

\end{frame}
