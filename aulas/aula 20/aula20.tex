\section{Introdução}

\begin{frame}[fragile]

  \frametitle{Questões da prova}

  Como armazenar a nota de cada uma\\
  de 10 questões da prova de {\bf um} aluno?


  \begin{exampleblock}{}
\begin{semiverbatim}\pause
float questoes[10];
int i;
\pause
for (i = 0; i < 10; i++) \{
    printf("Nota da questao \%d: ", i);
    scanf("\%f", \&questao[i]);
\}
\end{semiverbatim}
    \end{exampleblock}

  \pause

  \alert{E se fossem {\bf 100} alunos?}
\end{frame}

\begin{frame}[fragile]

  \frametitle{Cem alunos}

float questoes1[10], questoes2[10], \alert{\bf{}...}, questoes100[10];

    scanf("\%f", \&questao{\bf{}1}[i]);
\}
\pause
\alert{\bf{}...}
\pause
for (i = 0; i < 10; i++) \{
    printf("Nota da questao \%d do aluno {\bf{}100}: ", i);
    scanf("\%f", \&questao{\bf{}100}[i]);
\}
\end{semiverbatim}
    \end{exampleblock}

\end{frame}

 \frametitle{Cem alunos -- Melhorando}

 \begin{block}{Revendo vetores}

    \begin{itemize}

      \smallskip

      \smallskip

    \end{itemize}
 \end{block}
\end{frame}



\section{Matrizes}

\begin{frame}
  \frametitle{Vetores --- Definição}

  \begin{alertblock}{Matriz}
    Matriz é uma coleção de vetores.
  \end{alertblock}

  \pause

  \begin{block}{Características}
  \begin{itemize}
    \item É um vetor como outro qualquer
    \pause
    \item Cada vetor é acessado por por meio de um índice primário
    \pause
    \item Cada variável simples é acessada por por meio de um índice secundário
  \end{itemize}
 \end{block}
\end{frame}


\begin{frame}[fragile]

  \frametitle{Declarando uma matriz}

  \begin{block}{Matriz de tamanho $M \times N$}
    \alert{$<$tipo$>$} identificador [\alert{$<$linhas$>$}]  [\alert{$<$colunas$>$}]  {\bf ;}
  \end{block}

  \pause


  \begin{itemize}
  \item Uma matriz possui $linhas \times colunas$ variáveis do tipo \alert{$<$tipo$>$}
  \pause

  \item As linhas são numeradas de 0 a $linhas - 1$
  \pause

  \item As colunas são numeradas de 0 a $colunas - 1$

  \end{itemize}


  \pause

  \bigskip

  \begin{block}{Geralmente denotamos}
  \begin{itemize}
  \item o número de {\it{}linhas} por \alert{$m$}
  \item o número de {\it{}colunas} por \alert{$n$}
  \end{itemize}
  \end{block}



\end{frame}

\begin{frame}[fragile]

  \frametitle{Exemplo de declaração de matriz}

  \begin{exampleblock}{}
  \begin{verbatim}
int matriz [5][4];  \end{verbatim}
  \end{exampleblock}

  \vspace{20pt}

  \begin{center}
    \begin{tabular}{|l|c|c|c|c|}
      \cline{2-5}
      \multicolumn{1}{l|}{}&0&1&2&3\\ \hline
      0&&&&\\ \hline
      1&&&&\\ \hline
      2&&&&\\ \hline
      3&&&&\\ \hline
      4&&&&\\ \hline
    \end{tabular}
  \end{center}

\end{frame}

\begin{frame}

  \frametitle{Declarando uma vetor de múltiplas dimensões}

  \begin{block}{Vetor com $n$ dimensões}
    \alert{$<$tipo$>$} identificador
    [\alert{$<dim_1>$}]
    [\alert{$<dim_2>$}]
    \alert{$\dots$}
    [\alert{$<dim_n>$}]
  \end{block}

  \pause

  \begin{itemize}

  \item Possui $dim_1 \times dim_2 \times \dots \times \dim_N$
    variáveis do tipo \texttt{\alert{$<$tipo$>$}}

  \pause

  \item Cada dimensão é numerada de 0 a $dim_i - 1$

  \end{itemize}

\end{frame}


\begin{frame}[fragile]

  \frametitle{Acessando uma matriz}

    Em qualquer lugar onde você escreveria
    uma variável no seu programa, você pode usar
    um elemento de sua matriz, da seguinte forma:

    \pause

    \vspace{12pt}

    \begin{block}{}
      \texttt{nome\_da\_matriz} [\texttt{\alert{$<$linha$>$}}]
      [\texttt{\alert{$<$coluna$>$}}]
    \end{block}

    \vspace{12pt}
    \pause

    Ex: \verb% matriz[1][10]% --- Refere-se a variável na $2^a$
    linha e na $11^a$ coluna da matriz.

    \pause

  \begin{center}
  \alert{O compilador não verifica se você utilizou valores válidos para
    a linha e para a coluna.}
  \end{center}



\end{frame}


\section {Exemplos}

\begin{frame}[fragile]

  \frametitle{Exemplos com Matrizes}

  Lendo uma matriz $4 \times 5$ do teclado:

  \begin{exampleblock}{Lendo matriz}
   \begin{verbatim}
int i, j;
int matriz[4][5]

for (i = 0; i < 4; i++) {
    for (j = 0; j < 5; j++) {
        printf ("Valor da linha %d, coluna %d: ", i, j);
        scanf ("%d", &matriz[i][j]);
    }
}\end{verbatim}
  \end{exampleblock}

\end{frame}

\begin{frame}[fragile]

  \frametitle{Exemplos com Matrizes}

  Imprimindo uma matriz $4 \times 5$ do teclado:

  \begin{exampleblock}{Escrevendo uma matriz}
   \begin{verbatim}
int i, j;
int matriz[4][5]

for (i = 0; i < 4; i++) {
    printf("Linha %d: ", i);

    for (j = 0; j < 5; j++) {
        printf("%d ", matriz[i][j];
    }

    printf("\n");
}\end{verbatim}
  \end{exampleblock}
\end{frame}

\begin{frame}[containsverbatim]
  \frametitle{Matrizes e funções}

  \begin{itemize}
  \item Ao passar um {\bf vetor simples} como parâmetro, não é necessário fornecer o
    seu tamanho na declaração da função. 

  \item Quando o {\bf vetor é multi-dimensional} a possibilidade de não
    informar o tamanho na declaração se restringe à primeira dimensão
    apenas.
  \end{itemize}

  \begin{block}{}
\begin{verbatim}
void mostra_matriz(int mat[][10], int n_linhas) {
  ...
}
\end{verbatim}
  \end{block}

%  Veja o exemplo em {\tt matriz.c}.
\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\begin{frame}[containsverbatim]
  \frametitle{Matrizes e funções}

\begin{itemize}
\item Pode-se criar uma função deixando de indicar a primeira dimensão:
\begin{small}
\begin{verbatim}
void mostra_matriz(int mat[][10], int n_linhas) {
  ...
}
\end{verbatim}
\end{small}

\item Ou pode-se criar uma função indicando todas as dimensões:
\begin{small}
\begin{verbatim}
void mostra_matriz(int mat[5][10], int n_linhas) {
  ...
}
\end{verbatim}
\end{small}


\item Mas não pode-se deixar de indicar outras dimensões (exceto a primeira):
\begin{small}
\begin{verbatim}
void mostra_matriz(int mat[5][], int n_linhas) {
  //ESTE NÃO FUNCIONA
  ...
}
\end{verbatim}
\end{small}

\end{itemize}
%  Veja o exemplo em {\tt matriz.c}.
\end{frame}

\begin{frame}[containsverbatim]
\frametitle{Vetores multi-dimensionais em funções}

{\bf Lembre-se que vetores (multi-dimensionais ou não) são alterados quando passados como
parâmetro em uma função}

\begin{scriptsize}
\begin{verbatim}

void   teste (int mat[2][2]) {
  int i, j;

  for (i = 0; i < 2; i++) {
    for (j = 0; j < 2; j++){
         mat[i][j] = -1;
    }
  }
}

int main() {
  int mat[2][2] = { { 0, 1},
                    { 2, 3} };
		 
  teste(mat);
  //Neste ponto mat tem todas as suas posições com valor -1
  
  return 0;
}
\end{verbatim}
\end{scriptsize}
\end{frame}
